38       * Writes the font as an SVG file. This is from a segment of code from
39       * {@link SVGFont#main(String[])}. Warnings are suppressed to prevent Sonar
40       * from complaining about throwing an {@link Exception}.
41       * 
42       * @param ps
43       *            print stream to write to. (does not get closed)
44       * @param font
45       *            font to render
46       * @param id
47       *            ID of the font
48       * @param first
49       *            first character index to render, may be -1
50       * @param last
51       *            last character index to render, may be -1
52       * @param autoRange
53       *            find out the range automatically
54       * @param forceAscii
55       *            force ASCII range
56       * @throws Exception
57       */
58      @SuppressWarnings("all")
59      public static void writeFontAsSvg(final PrintStream ps, final Font font,
60              @Nullable final String id, final int first, final int last,
61              final boolean autoRange, final boolean forceAscii) throws Exception {
62          writeSvgBegin(ps);
63          writeSvgDefsBegin(ps);
64          writeFontAsSVGFragment(ps, font, id, first, last, autoRange, forceAscii);
65          writeSvgDefsEnd(ps);
66          writeSvgEnd(ps);
67      }
